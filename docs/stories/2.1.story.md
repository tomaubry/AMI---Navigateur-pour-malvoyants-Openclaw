# Story 2.1 — App Client : Bouton Micro + WebSocket + Vue noVNC

## User Story

**En tant que** développeur,
**Je veux** créer une app web légère avec un bouton "Parler à Ami" qui capture le micro et stream l'audio vers le VPS via WebSocket,
**Afin que** l'utilisateur démarre une conversation vocale en un clic, sans numéro de téléphone.

---

## Critères d'Acceptation

1. **Étant donné** que l'app est ouverte dans un navigateur, **quand** je clique sur "Parler à Ami", **alors** le micro est capturé et un WebSocket s'ouvre vers `wss://VPS/audio`.
2. **Étant donné** que le WebSocket est ouvert, **quand** je parle, **alors** le VPS reçoit des chunks audio PCM 16kHz.
3. **Étant donné** que le VPS envoie de l'audio en retour (TTS), **quand** les chunks arrivent, **alors** le navigateur les joue via `AudioContext`.
4. **Étant donné** que la conversation est en cours, **alors** l'app affiche le flux noVNC (Chromium naviguant en direct) dans la même interface.
5. **Étant donné** que je clique sur "Arrêter", **alors** le micro et le WebSocket sont fermés proprement.

  ```javascript
  import 'dotenv/config';
  import express from 'express';
  import { createServer } from 'http';
  import { WebSocketServer } from 'ws';
  import path from 'path';
  import { fileURLToPath } from 'url';

  const __dirname = path.dirname(fileURLToPath(import.meta.url));
  const app = express();

  // Sert l'app front-end
  app.use(express.static(path.join(__dirname, '../../app')));

  const server = createServer(app);

  // WebSocket audio : reçoit le micro de l'app, renvoie le TTS
  const wss = new WebSocketServer({ server, path: '/audio' });

  wss.on('connection', (ws) => {
    console.log('[WS] Nouvelle session audio connectée');

    ws.on('message', (data) => {
      // data = Buffer audio PCM 16kHz mono depuis getUserMedia()
      console.log('[Audio] chunk reçu, taille:', data.length);
      // Prochaines stories : passer à Deepgram STT
    });

    ws.on('close', () => console.log('[WS] Session fermée'));
    ws.on('error', (err) => console.error('[WS] Erreur:', err.message));
  });

  server.listen(process.env.PORT || 4000, () => {
    console.log(`Orchestrateur Ami démarré sur :${process.env.PORT || 4000}`);
  });
  ```

- [x] **2.1.5** Créer `app/app.js` — logique WebSocket + micro :
  ```javascript
  let ws = null;
  let mediaStream = null;
  let audioCtx = null;
  let processor = null;
  let active = false;

  const WS_URL = `wss://${location.host}/audio`;

  async function toggleConversation() {
    active ? stopConversation() : await startConversation();
  }

  async function startConversation() {
    try {
      mediaStream = await navigator.mediaDevices.getUserMedia({ audio: { sampleRate: 16000, channelCount: 1 } });
      ws = new WebSocket(WS_URL);
      ws.binaryType = 'arraybuffer';

      ws.onopen = () => {
        setStatus('Connecté — parlez maintenant');
        setupAudioCapture();
      };

      ws.onmessage = (e) => playAudioChunk(e.data);
      ws.onclose = () => { stopConversation(); setStatus('Déconnecté'); };
      ws.onerror = () => setStatus('Erreur de connexion');

      setActive(true);
    } catch (err) {
      setStatus('Micro refusé : ' + err.message);
    }
  }

  function setupAudioCapture() {
    audioCtx = new AudioContext({ sampleRate: 16000 });
    const source = audioCtx.createMediaStreamSource(mediaStream);
    processor = audioCtx.createScriptProcessor(4096, 1, 1);

    processor.onaudioprocess = (e) => {
      if (ws?.readyState !== WebSocket.OPEN) return;
      const pcm = e.inputBuffer.getChannelData(0);
      const int16 = new Int16Array(pcm.length);
      for (let i = 0; i < pcm.length; i++) int16[i] = Math.max(-32768, Math.min(32767, pcm[i] * 32768));
      ws.send(int16.buffer);
    };

    source.connect(processor);
    processor.connect(audioCtx.destination);
  }

  function playAudioChunk(arrayBuffer) {
    if (!audioCtx) audioCtx = new AudioContext({ sampleRate: 16000 });
    audioCtx.decodeAudioData(arrayBuffer.slice(0)).then((buf) => {
      const src = audioCtx.createBufferSource();
      src.buffer = buf;
      src.connect(audioCtx.destination);
      src.start();
    }).catch(() => {});
  }

  function stopConversation() {
    processor?.disconnect();
    audioCtx?.close();
    mediaStream?.getTracks().forEach(t => t.stop());
    ws?.close();
    ws = null; mediaStream = null; audioCtx = null; processor = null;
    setActive(false);
    setStatus('Cliquez pour démarrer');
  }

  function setActive(val) {
    active = val;
    const btn = document.getElementById('btn');
    btn.textContent = val ? 'Arrêter' : 'Parler à Ami';
    btn.className = 'btn-talk ' + (val ? 'active' : 'idle');
  }

  function setStatus(msg) { document.getElementById('status').textContent = msg; }

  function addMessage(role, text) {
    const div = document.createElement('div');
    div.className = 'msg ' + role;
    div.textContent = (role === 'ami' ? '🤖 Ami : ' : '👤 Vous : ') + text;
    document.getElementById('transcript').appendChild(div);
    div.scrollIntoView();
  }
  ```

## Notes Dev

- **getUserMedia sampleRate :** Certains navigateurs ignorent le `sampleRate: 16000` et capturent à 48kHz. Si c'est le cas, il faudra ré-échantillonner côté client ou côté serveur (librairie `node-resample` ou simple resampling par interpolation).
- **AudioContext** : Doit être créé après un geste utilisateur (clic) — c'est déjà le cas avec notre bouton.
- **WSS (WebSocket sécurisé) :** Nginx upgrade HTTP → HTTPS/WSS. `getUserMedia` ne fonctionne qu'en HTTPS (sauf `localhost`). En prod, assurer un certificat TLS valide sur Nginx.
- **noVNC dans iframe :** Le `autoconnect=true&resize=scale` permet la connexion automatique et l'adaptation à la taille du panel.
- **Sécurité :** Ajouter un token d'authentification dans la query string du WebSocket en production (`wss://host/audio?token=xxx`).
